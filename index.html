Usage

Fairy fairy = Fairy.create();
Person person = fairy.person();

person.getFullName();        // Chloe Barker
person.getEmail();           // barker@yahoo.com
person.getTelephoneNumber(); // 690-950-802
person.getJobTitle();        // Software Developer

Supported locales

EN, PL, DE, FR, ES, SV, ZH, KA, IT, BR, SK

Fairy plFairy = Fairy.create(Locale.forLanguageTag("pl"));
Fairy brFairy = Fairy.create(Locale.forLanguageTag("br"));
